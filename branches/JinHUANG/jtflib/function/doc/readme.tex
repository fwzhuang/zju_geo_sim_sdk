\subsection{A Simple Case}
\begin{equation}
  \begin{split}
    x&\in R^4\\
    f(x) &=
    \begin{pmatrix}
      f_0(x)\\
      f_1(x)
    \end{pmatrix}\\
    f_0(x) &= 10
    \begin{pmatrix}
      \cos(x_0)\\
      x_1x_2\\
    \end{pmatrix}\\
    f_1(x) &=
    \begin{pmatrix}
      1-x_1\\
      x_4-x_0\\
      x_2-\sin(x_0)
    \end{pmatrix}
  \end{split}
\end{equation}
